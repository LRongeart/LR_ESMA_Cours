	<script>
		function darkMode() {
			let element = document.body;
			let content = document.getElementById("DarkModetext");
			element.className = "dark-mode";
			content.innerText = "Dark Mode is ON";
		}
		function lightMode() {
			let element = document.body;
			let content = document.getElementById("DarkModetext");
			element.className = "light-mode";
			content.innerText = "Dark Mode is OFF";
		}
